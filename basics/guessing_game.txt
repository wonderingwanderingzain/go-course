package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {

	// generating random number....learn it later
	source := rand.NewSource(time.Now().UnixNano())
	random := rand.New(source)

	// Generate a random number between 1 and 100
	target := random.Intn(100) + 1

	// Welcome message
	fmt.Println("Welcome to The Guessing Game!")
	fmt.Println("I have chosen a number between 1 and 100")
	fmt.Println("Guess what it is?")

	var guess int
	var tries int = 1 // number on which user guessed correct
	for {

		fmt.Println("Guess the number : ")
		fmt.Scanln(&guess)

		// check if guess is correct
		if guess == target {

			if tries == 1 { // if guessed on 1st try :)
				fmt.Println("Congratulations! You guessed the correct number! It is ", target, ". And you guessed it on ", tries, "st try ğŸ™ƒ")
				break
			}

			fmt.Println("Congratulations! You guessed the correct number! It is ", target, ". And you guessed it in ", tries, " tries ğŸ™ƒ")
			break

		} else if guess < target {
			fmt.Println("Low guess......try a higher number")
		} else {
			fmt.Println("High guess....try a lower number")
		}

		tries++

	}

}
