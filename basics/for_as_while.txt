package main

import "fmt"

func main() {

	i := 1

	for i < 6 {
		fmt.Println(" Iteration : ", i)
		i++
	}

	// for as while with break
	sum := 0
	for {
		sum += 10
		fmt.Println("Sum : ", sum)
		if sum >= 50 {
			break
		}
	}

	// for as while with continue
	j := 1
	for j <= 10 {
		if j%2 == 0 {
			j++
			continue
		}
		fmt.Println("Odd number is : ", j)
		j++
	}

}
